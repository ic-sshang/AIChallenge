openapi: 3.0.0
info:
  title: ICChaseScreeningAPI
  description: |-
    The ICChaseScreeningAPI provides a secure interface for onboarding merchants onto the IC platform, managing their profiles, and facilitating screening and approval through integration with the JPM Onboarding System.
  version: 1.0.0
  contact:
    name: API Support
    email: test@test.com
security:
  - bearerAuth: []
tags:
  - name: merchant
    description: Operations related to merchant management
  - name: screening
    description: Operations related to merchant screening and approval
paths:
  /merchants:
    post:
      tags:
        - merchant
      summary: Create a new merchant
      requestBody:
        required: true
        content:
          application/json:
            schema:
              type: object
              properties:
                merchant:
                  type: object
                  properties:
                    type:
                      type: string
                      enum: ["ORGANIZATION", "INDIVIDUAL"]
                      description: The type of the merchant entity
                    email:
                      type: string
                      format: email
                      maxLength: 320
                      description: The merchant's contact email
                    roles:
                      type: array
                      items:
                        type: string
                        enum: ["CLIENT", "CONTROLLER", "BENEFICIAL_OWNER", "DIRECTOR", "PRIMARY_CONTACT","DECISION_MAKER", "AUTHORIZED_USER"]
                      description: Roles assigned to the merchant
                    organization:
                      type: object
                      properties:
                        type:
                          type: string
                          enum: ["LIMITED_LIABILITY_PARTNERSHIP","LIMITED_PARTNERSHIP", "GENERAL_PARTNERSHIP","LIMITED_LIABILITY_COMPANY","C_CORPORATION","S_CORPORATION","PARTNERSHIP","PUBLICLY_TRADED_COMPANY","NON_PROFIT_CORPORATION","GOVERNMENT_ENTITY","SOLE_PROPRIETORSHIP","UNINCORPORATED_ASSOCIATION"]
                          description: The legal structure of the organization
                        name:
                          type: string
                          maxLength: 500
                          description: The name of the organization
                        country:
                          type: string
                          pattern: '^[A-Z]{2}$'
                          description: The country of formation (ISO 3166-1 alpha-2 code)
                      required:
                        - type
                        - name
                        - country
                      description: Details of the merchant's organization
                    contacts:
                      type: array
                      items:
                        type: object
                        properties:
                          type:
                            type: string
                            enum: ["INDIVIDUAL"]
                            description: The type of the contact
                          email:
                            type: string
                            format: email
                            maxLength: 320
                            description: The contact's email
                          roles:
                            type: array
                            items:
                              type: string
                              enum: ["CONTROLLER", "BENEFICIAL_OWNER", "DIRECTOR", "PRIMARY_CONTACT", "DECISION_MAKER", "AUTHORIZED_USER"]
                            description: Roles assigned to the contact
                          firstName:
                            type: string
                            maxLength: 30
                            description: The first name of the individual
                          lastName:
                            type: string
                            maxLength: 30
                            description: The last name of the individual
                          country:
                            type: string
                            pattern: '^[A-Z]{2}$'
                            description: The country of residence (ISO 3166-1 alpha-2 code)
                        required:
                          - type
                          - email
                          - roles
                          - firstName
                          - lastName
                          - country
                        description: An individual associated with the merchant
                      description: List of contacts for the merchant
                    products:
                      type: array
                      items:
                        type: string
                        enum: ["EMBEDDED_PAYMENTS"]
                      description: Products the merchant is signing up for
                  required:
                    - type
                    - email
                    - roles
                    - organization
                    - contacts
                    - products
                  description: The merchant entity
              required:
                - merchant
            example:
              merchant:
                type: "ORGANIZATION"
                email: "monica@cpgetaways.com"
                roles: ["CLIENT"]
                organization:
                  type: "SOLE_PROPRIETORSHIP"
                  name: "Central Park Getaways"
                  country: "US"
                contacts:
                  - type: "INDIVIDUAL"
                    email: "monica@cpgetaways.com"
                    roles: ["CONTROLLER", "BENEFICIAL_OWNER"]
                    firstName: "Monica"
                    lastName: "Gellar"
                    country: "US"
                products:
                  - "EMBEDDED_PAYMENTS"
      responses:
        '201':
          description: Merchant created
          content:
            application/json:
              schema:
                type: object
                properties:
                  id:
                    type: string
                  createdAt:
                    type: string
              example:
                id: "1234"
                createdAt: "2025-03-27T12:00:00Z"
        '400':
          description: Invalid Request
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/Error'
        '401':
          description: Unauthenticated
        '500':
          description: Unexpected error
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/Error'
  /merchants/{merchantId}:
    get:
      tags:
        - merchant
      summary: Retrieve a merchant by ID
      parameters:
        - name: merchantId
          in: path
          required: true
          schema:
            type: string
          description: The unique identifier of the merchant
      responses:
        '200':
          description: Merchant details retrieved successfully
          content:
            application/json:
              schema:
                type: object
                properties:
                  id:
                    type: string
                  emailAddress:
                    type: string
                  merchantDBA:
                    type: string

                    type: string
              example:

                createdAt: "2025-03-27T12:00:00Z"

                $ref: '#/components/schemas/Error'

                $ref: '#/components/schemas/Error'

              schema:
