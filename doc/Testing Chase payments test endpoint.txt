This page contains 3 PowerShell scripts to send GET request to JPMorgan payments API. You can use those for testing of Chase authentication.
Sending the payment request without JWT token:
powershell
# Define the URL
$url = "<https://api-mock.payments.jpmorgan.com/api/v2/payments?requestIdentifier=14cc0270-7bed-11e9-a188-1763956dd7f6>"

# Define headers
$headers = @{
    "Authorization" = "Bearer "
    "Accept" = "application/json"
    "merchant-id" = "998482157632"
    "minorVersion" = ""
    "platform-id" = "909876543210"
    "request-id" = "10cc0270-7bed-11e9-a188-1763956dd7f6"
}

# Send the request and capture the response
try {
    $response = Invoke-RestMethod -Uri $url -Method Get -Headers $headers -ErrorAction Stop

    Write-Host "`n--- Response Summary ---"
    Write-Host "Status: 200 OK (assumed from successful Invoke-RestMethod)"
    Write-Host "`n--- Headers ---"
    $headers.GetEnumerator() | ForEach-Object { Write-Host "$($_.Key): $($_.Value)" }

    Write-Host "`n--- Raw Response ---"
    $responseRaw = $response | Out-String
    Write-Output $responseRaw

    Write-Host "`n--- Parsed JSON ---"
    $response | ConvertTo-Json -Depth 10 | Write-Output
}
catch {
    Write-Host "`n--- ERROR ---"
    Write-Host "Exception Message: $($_.Exception.Message)"

    if ($_.Exception.Response -ne $null) {
        $reader = New-Object System.IO.StreamReader($_.Exception.Response.GetResponseStream())
        $reader.BaseStream.Position = 0
        $reader.DiscardBufferedData()
        $errorBody = $reader.ReadToEnd()

        Write-Host "`n--- Error Response Body ---"
        Write-Output $errorBody
    }
}

If you want to compare the payment request with and without JWT token, use the following script:
powershell
# Define the URL
$url = "<https://api-mock.payments.jpmorgan.com/api/v2/payments?requestIdentifier=14cc0270-7bed-11e9-a188-1763956dd7f6>"

# Define the JWT token
$jwtToken = "eyJ0eXAiOiJKV1QiLCJhbGciOiJSUzI1NiIsIng1dCI6IjlCY3doSUVhaml0bWZQUERlY3E0UGdFVUo0OCIsImtpZCI6IjlCY3doSUVhaml0bWZQUERlY3E0UGdFVUo0OCJ9..."

# Function to send request and return response
function Invoke-ApiRequest {
    param (
        [string]$tokenLabel,
        [string]$tokenValue
    )

    $headers = @{
        "Authorization" = "Bearer $tokenValue"
        "Accept" = "application/json"
        "merchant-id" = "998482157632"
        "minorVersion" = ""
        "platform-id" = "909876543210"
        "request-id" = "10cc0270-7bed-11e9-a188-1763956dd7f6"
    }

    try {
        $response = Invoke-RestMethod -Uri $url -Method Get -Headers $headers -ErrorAction Stop
        return @{ Label = $tokenLabel; Success = $true; Response = $response }
    }
    catch {
        return @{ Label = $tokenLabel; Success = $false; Error = $_.Exception.Message }
    }
}

# Run both requests
$responseNoToken = Invoke-ApiRequest -tokenLabel "No Token" -tokenValue ""
$responseWithToken = Invoke-ApiRequest -tokenLabel "With Token" -tokenValue $jwtToken

# Compare results
Write-Host "`n--- COMPARISON ---"

foreach ($resp in @($responseNoToken, $responseWithToken)) {
    Write-Host "`n[$($resp.Label)]"
    if ($resp.Success) {
        Write-Host "Status: 200 OK"
        Write-Host "Response:"
        $resp.Response | ConvertTo-Json -Depth 10 | Write-Output
    } else {
        Write-Host "Request Failed:"
        Write-Host $resp.Error
    }
}
