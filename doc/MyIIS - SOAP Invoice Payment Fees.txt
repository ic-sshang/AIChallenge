HTTP POST Request
Use a simple c# HTTP client and POST instead of adding a generated service reference proxy.  The operation and XML payloads are not complex.  
Use the value of the JWT token that was received in the authorization header as the value for the element soap12:Envelope/soap12:Header/BearerTokenHeader/BearerToken.
 SOAP API definition as of 2/23/2023
text
POST /portal/webservices/invoicepaymentfees.asmx HTTP/1.1
Host: devtest4.invoicecloud.com
Content-Type: application/soap+xml; charset=utf-8
Content-Length: length

<?xml version="1.0" encoding="utf-8"?>
<soap12:Envelope xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:xsd="http://www.w3.org/2001/XMLSchema" xmlns:soap12="http://www.w3.org/2003/05/soap-envelope">
  <soap12:Header>
    <BearerTokenHeader xmlns="https://www.invoicecloud.com/portal/webservices/InvoicePaymentFees/">
      <BearerToken>string</BearerToken>
    </BearerTokenHeader>
  </soap12:Header>
  <soap12:Body>
    <CalculateSingleInvoicePaymentFee xmlns="https://www.invoicecloud.com/portal/webservices/InvoicePaymentFees/">
      <request>
        <InvoiceGuid>guid</InvoiceGuid>
        <PaymentMethod>
          <PaymentAmount>double</PaymentAmount>
          <PaymentSourceId>int</PaymentSourceId>
          <Creditcard>
            <BIN>string</BIN>
            <PAN>string</PAN>
            <ExpMonth>int</ExpMonth>
            <ExpYear>int</ExpYear>
          </Creditcard>
          <ACH>
            <RTE>string</RTE>
            <AccountNumber>string</AccountNumber>
          </ACH>
          <PaypalBrands>
            <PaymentTypeId>int</PaymentTypeId>
          </PaypalBrands>
        </PaymentMethod>
      </request>
    </CalculateSingleInvoicePaymentFee>
  </soap12:Body>
</soap12:Envelope>
