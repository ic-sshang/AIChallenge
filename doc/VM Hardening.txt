Linux CrowdStrike Agent Installation
IP=<IP of VM>
USERNAME=<Name of User>
CID=<Get From KeyVault>

az ssh config --ip $IP --file ./azure-sshconfig --keys-destination-folder ./azure-keys

scp -F ./azure-sshconfig falcon-sensor_6.54.0-15110_amd64.deb $USERNAME@invoicecloud.com@$IP:/home/$USERNAME

sudo dpkg -i falcon-sensor_6.54.0-15110_amd64.deb

sudo apt --fix-broken install

sudo /opt/CrowdStrike/falconctl -s --cid=$CID

sudo service falcon-sensor start

sudo systemctl start falcon-sensor

ps -e | grep falcon-sensor
