  $('#UsersGrid').DataTable({
    ajax: '/api/users',
    columns: [
      { data: 'name' },
      { data: 'email' },
      {
        data: null,
        render: function (data, type, row) {
          return `<button class="btn-delete" data-id="${row.id}">Delete</button>`;
        }
      }
    ],
    serverSide: true,
    processing: true
  });
});
