The standard guideline for building an activity diagram is to show the flow and logic of an application.
The starting and end points are used to indicate where your diagram starts and ends. Activity nodes are used to explain what at each step of the logic.

class Program
{
	static void Main(string[] args)
    {
		int terminalID = Convert.ToInteger(args[0]);
		Repository repository = new Repository();
		var terminal = repository.SelTerminal(terminalID: terminalID);
		
		switch(terminal.PaymentType)
		{
			case PaymentType.CreditCard:
				CCProcessor ccProcessor = new CCProcessor();
				ccProcessor.Run(terminal.TerminalID);
				break;
			case PaymentType.ACH:
				ACHProcessor achProcessor = new ACHProcessor();
				achProcessor.Run(terminal.TerminalID);
				break;
			case PaymentType.OBD:
				OBDProcessor obdProcessor = new OBDProcessor();
				obdProcessor.Run(terminal.TerminalID);
				break;
			default:
				throw NotImplementedException("Given payment type not implemented.");
		}
		
		return;
	}
}
