Here’s a full list of every version of every package in every one of our Azure Artifacts feeds.
Script to collect this data:
from azure.devops.connection import Connection
from msrest.authentication import BasicAuthentication
import pprint
from os import environ

# Personal Access Token (PAT) and organization URL
personal_access_token = environ['AZURE_DEVOPS_PAT']
organization_url = 'https://dev.azure.com/InvoiceCloud'

# Create a connection to the organization
credentials = BasicAuthentication('', personal_access_token)
connection = Connection(base_url=organization_url, creds=credentials)

# Get a client (the "core" client provides access to projects, teams, etc.)
core_client = connection.clients.get_core_client()
feed_client = connection.clients_v7_1.get_feed_client()

# List all feeds in the organization
feeds = feed_client.get_feeds()

# Dictionary to hold package versions
package_versions = {}

print("Project,Feed,View,Package,Version")

# Iterate through feeds
for feed in feeds:
    if feed.project:
        packages = feed_client.get_packages(feed_id=feed.id, project=feed.project.id)
    else:
        packages = feed_client.get_packages(feed_id=feed.id)
    
    # Iterate through packages in the feed
    for package in packages:
        # Get the latest version of the package
        if feed.project:
            package_versions = feed_client.get_package_versions(feed_id=feed.id, package_id=package.id, project=feed.project.id)
        else:
            package_versions = feed_client.get_package_versions(feed_id=feed.id, package_id=package.id)
        for version in package_versions:
            for view in version.views:
                print(f"{feed.project.name if feed.project else 'n/a'},{feed.name},{view.name},{package.name},{version.version}")
