    const selected = {};

    function selectChanged(chk) {
        const invoiceGUID = chk.getAttribute('data-invoice');
        if (!invoiceGUID) return;

        if (chk.checked) {
            selected[invoiceGUID] = true;
        } else {
            delete selected[invoiceGUID];
        }

        document.getElementById('hdnSelectedRowsData').value = JSON.stringify(selected);
    }

    // DOMContentLoaded works on non-AJAX pages. If there’s any chance of partial postbacks, use MS AJAX hooks so your inits run every time the DOM is refreshed.
    (function () {
      function init() {
        if (typeof init_client_validate === 'function') init_client_validate();
        if (typeof init_date_picker === 'function') init_date_picker();
      }
    
      // If MS AJAX is present (ScriptManager on the page), hook its lifecycle
      if (window.Sys && Sys.Application) {
        Sys.Application.add_load(init); // initial load + every partial postback
      } else {
        // Plain WebForms / no partial postbacks
        document.addEventListener('DOMContentLoaded', init);
      }
    })();;
</script>
