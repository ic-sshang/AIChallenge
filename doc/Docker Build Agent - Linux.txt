Clean VM
To clean the directories when it starts having out of memory issues run the following commands:
bash
sudo docker system prune -a -f --volumes

sudo apt-get autoremove

sudo du -sh /var/cache/apt

sudo apt-get clean

sudo journalctl --vacuum-time=3d

rm -rf ~/.cache/thumbnails/*

IN_USE=$(uname -r)
OLD_KERNELS=$(dpkg --get-selections | grep -vE "linux-headers-generic|linux-image-generic" | grep -v "${IN_USE%%-generic}" | grep -Ei 'linux-image|linux-headers|linux-modules' | awk '{print $1}')
sudo apt-get purge $OLD_KERNELS
