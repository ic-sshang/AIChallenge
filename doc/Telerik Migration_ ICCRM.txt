Update instances of radGrid to datatables following the installation guide: 
https://datatables.net/manual/installation
 
Ensure the following resources are present: 
xml
<script type="text/javascript" charset="utf8" src="resources/plugins/datatables/jquery.dataTables.js"></script>
<script src="resources/js/ic-custom-datatable.js"></script>
<link rel="stylesheet" type="text/css" href="resources/plugins/datatables/jquery.dataTables.css">
<link href="resources/css/custom-datatable.css" rel="stylesheet"/>
<link href="resources/css/tools-datatable.css" rel="stylesheet"/>
Convert table to standard markup:
<table id="dgData" class="display" style="width:100%">
  <thead>
    <tr>
      <!-- <th>Biller DBA</th> Uncomment if needed -->
      <th>Biller ID</th>
      <th>Request ID</th>
      <th>Certegy Reference</th>
      <th>RTE - DDA</th>
      <th>Response Code</th>
      <th>Request Date</th>
    </tr>
  </thead>
</table>
Initialize datatables:
xml
<script src="https://code.jquery.com/jquery-3.7.1.js"></script>
<script src="https://cdn.datatables.net/2.0.7/js/dataTables.min.js"></script>
<link rel="stylesheet" href="https://cdn.datatables.net/2.0.7/css/dataTables.min.css">

<!-- Optional export buttons -->
<script src="https://cdn.datatables.net/buttons/3.0.1/js/dataTables.buttons.min.js"></script>
<script src="https://cdn.datatables.net/buttons/3.0.1/js/buttons.html5.min.js"></script>
<script src="https://cdn.datatables.net/buttons/3.0.1/js/buttons.print.min.js"></script>
<script src="https://cdnjs.cloudflare.com/ajax/libs/jszip/3.10.1/jszip.min.js"></script>
<link rel="stylesheet" href="https://cdn.datatables.net/buttons/3.0.1/css/buttons.dataTables.min.css">

<script>
$(document).ready(function () {
    $('#dgData').DataTable({
        ajax: '/api/data-endpoint',  // Replace with your actual data URL
        columns: [
            { data: 'BillerID' },
            { data: 'RequestID' },
            { data: 'CertegyReference' },
            { data: 'RTEDDA' },
            { data: 'ResponseCode' },
            { 
              data: 'RequestedDate',
              render: function(data, type, row) {
                  return new Date(data).toLocaleString(); // Format as MM/DD/YYYY hh:mm:ss
              }
            }
        ],
        pageLength: 20,
        lengthChange: false,
        ordering: true,
        dom: 'Bfrtip',
        buttons: [
            {
                extend: 'excelHtml5',
                title: 'ACH Verification API Transactions Report',
                filename: 'ACH_Verification_API_Transactions_Report',
                exportOptions: {
                    columns: ':visible'
                }
            }
        ]
    });
});
</script>
